  summarize(n_obs = n())%>%

instacart %>%
  group_by(aisle) %>%
  summarize(n_obs = n())%>%
  arrange(desc(n_obs))%>%
  mutate(
        aisle = factor(aisle),
        aisle = fct_reorder(aisle, n_obs)
    ) %>%
  mutate(text_label = str_c("\nAisle ID: $", aisle, "\nNumber of items: ",n_obs)) %>% 
  plot_ly(
    x = ~aisle, y = ~n_obs ,color = ~aisle, type = "bar", colors = "viridis",
     text = ~text_label, alpha = 0.8)
